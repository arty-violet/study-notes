"Заявления"

----------------------------------

+-

if

Условная конструкция if указывает блок кода, который будет выполнен, если условие истинно

if (условие) {
  // Инструкции, которые выполнятся, если условие истинно;
}
if (element.classList.contains('some-class')) {
  console.log('Условие истинно');
  // Сообщение выведется, только если класс у элемента есть (возвращает true).
}

else

Условная конструкция else указывает блок кода, который будет выполнен, если условие ложно

if(условие) {
  // Инструкции, которые выполнятся, если условие истинно;
} else {
  // Инструкции, которые выполнятся, если условие ложно;
}

else if

Конструкция else if позволяет добавить в условную конструкцию альтернативную ветку c условием.

if (условие-1) {
  // Инструкции выполнятся, если истинно условие-1
} else if (условие-2) {
  // Инструкции выполнятся, если условие-1 ложно, а условие-2 истинно
} else {
  // Инструкции выполнятся, если оба условия ложны
}

for ... of

Цикл for of — это конструкция, которая позволяет выполнить код несколько раз.

for (let переменная of коллекция) {
  // Код, который нужно выполнить несколько раз
}

* При создании цикла в круглых скобках также нужно указать переменную. Обычно для этого объявляют новую переменную и используют её только внутри цикла

let elements = document.querySelectorAll('selector'); // Находим все нужные селекторы
for (let element of elements) { // Создаём цикл и переменную
  console.log(element); // Выводим элементы в консоль
}
for (let element of elements) { // Создаём цикл и переменную
  element.событие = function () {}; // Добавляем элементам обработчик(и) события
}
